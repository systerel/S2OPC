62// A&C instance structure

64struct _SOPC_AlarmCondition
65{
66 SOPC_Mutex mut; // Mutex to protect the A&C content during modification / access
67 SOPC_NodeId notifierNode; // Id of the node used to trigger events in address space
68 SOPC_NodeId conditionNode; // Id of the A&C node
69 int32_t autoRetainFlag; // Flag to manage Retain state automatically
70 int32_t autoAckOnActiveFlag; // Flag to set automatically the acknowledgeable state on active state transition
71 int32_t autoConfOnAckedFlag; // Flag to set automatically the confirmable state on acknowledged state transitions
72 SOPC_Event* data; // Stores event field states of the A&C, those are synchronized with address space node variables
73 SOPC_EventIdList* ackEventIds; // List of EventIds recorded for AckedState since last change to False state
74 SOPC_EventIdList* confEventIds; // List of EventIds recorded for ConfirmedState since last change to False state
75 SOPC_EventIdList* globalEventIds; // List of latest EventIds recorded
76
77 SOPC_Event* nodeIds; // Stores the node ids of corresponding nodes for each event fields
78 SOPC_NodeId* methodIds; // Stores the nodes ids of the well-known and supported A&C methods
79 SOPC_Dict*
80 varChangedCbDict; // Stores the callback for variable changes (path -> SOPC_AlarmCondition_StateChanged_Fct_Ctx)
81};
