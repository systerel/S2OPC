20 /* This module implements MQTT Transport Protocol */
21 
22 /* To use this module, you can follow the following steps (SYNCHRONE API with callback of reception). For more examples,
23  * see mqtt_pub_test.c.*/
24 
25 /* 1) Instantiate a MQTT Manager with SOPC_MQTT_MGR_Create : MqttManagerHandle * pMgr ; SOPC_MQTT_MGR_Create(&pWks);*/
26 /* 2) Define a reception callback : static void cbReceivedMessage(MqttTransportHandle* pCtx, uint8_t* data, uint16_t
27  * size){ return; } */
28 /* 3) Get a new transport handle with SOPC_MQTT_TRANSPORT_SYNCH_GetHandle : MqttTransportHandle *pHandleSynchroContext =
29  * SOPC_MQTT_TRANSPORT_SYNCH_GetHandle(pWks, "tcp://192.168.56.1", cbReceivedMessage); */
30 /* 4) Send a message with SOPC_MQTT_TRANSPORT_SYNCH_SendMessage :
31  * SOPC_MQTT_TRANSPORT_SYNCH_SendMessage(pHandleSynchroContext, buffer, bufferSizeInBytes, 2000); */
32 /* 5) On message reception, your callback "cbReceivedMessage" is called. */
33 /* 6) Release your transport handle with SOPC_MQTT_TRANSPORT_SYNCH_ReleaseHandle :
34  * SOPC_MQTT_TRANSPORT_SYNCH_ReleaseHandle(&pHandleSynchroContext).*/
