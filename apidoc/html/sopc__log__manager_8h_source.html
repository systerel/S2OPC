48 /*
49  * \brief Creates a new log file and log instance and prints the starting timestamp
50  *
51  * \param logDirPath Absolute or relative path of the directory to be used for logs (shall exist and terminate with
52  * directory separator)
53  * \param logFileName The file name to be used without extension. A prefix, an integer suffix and extension will be
54  * added automatically.
55  * \param category A category name if the log file is used for several categories or NULL. Truncated if more than 9
56  * characters.
57  * \param maxBytes A maximum amount of bytes by log file before opening a new file incrementing the integer suffix.
58  * It is a best effort value (amount verified after each print).
59  * \param maxFiles A maximum number of files to be used, when reached the older log file is overwritten
60  * (starting with *_00001.log)
61  *
62  * \return The log instance to be used to add traces
63  * */

66  const char* logFileName,
67  const char* category,
68  uint32_t maxBytes, // New file created when maxBytes reached (after printing latest trace)
69  uint16_t maxFiles); // Old logs overwritten when maxFiles reached
