61 /*
62  * \brief Description of an encodeable type field.
63  *
64  * This structure has been designed to be very compact and fit into 8 bytes.
65  * The \c isBuiltIn field indicates whether the field type is built-in or
66  * defined in the address space. In the first case, the type index is a member
67  * of the \c SOPC_BuiltinId enumeration. In the second case, it is a member of
68  * the \c SOPC_TypeInternalIndex enumeration.
69  *
70  * The \c isArrayLength field indicates whether this field contains the length
71  * of an array. When true, the field must be of built-in type \c Int32, and the
72  * array is described by the next field.
73  *
74  * The \c isToEncode field indicates whether this field shall be encoded and
75  * decoded. When false, the field is only initialized and cleared.
76  *
77  * Finally, the \c offset field gives the offset in bytes of the field in the
78  * object structure.
79  */
80 typedef struct SOPC_EncodeableType_FieldDescriptor
81 {
82  bool isBuiltIn : 1;
83  bool isArrayLength : 1;
84  bool isToEncode : 1;
85  uint32_t typeIndex : 29;
86  uint32_t offset;
87 } SOPC_EncodeableType_FieldDescriptor;
