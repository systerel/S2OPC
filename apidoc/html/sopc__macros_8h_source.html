70/*
71 * This macro is used to mimic the GNU TEMP_FAILURE_RETRY macro for environments
72 * that don't have the GNU C library.
73 *
74 * See https://www.gnu.org/software/libc/manual/html_node/Interrupted-Primitives.html for details.
75 *
76 * See also https://man7.org/linux/man-pages/man7/signal.7.html for the expected behaviour under Linux.
77 */
78#define S2OPC_TEMP_FAILURE_RETRY(result, exp) \
79 do \
80 { \
81 do \
82 { \
83 result = (int) (exp); \
84 } while ((result == -1) && (errno == EINTR)); \
85 } while (false);
